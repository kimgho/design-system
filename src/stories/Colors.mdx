import { Meta, ColorPalette, ColorItem } from '@storybook/addon-docs/blocks';
import { colorTokens } from '../theme/tokens/colors';

export const getColors = colorObj => {
  return Object.keys(colorObj).reduce((acc, key) => {
    if (colorObj[key].value) {
      acc[key] = colorObj[key].value;
    }
    return acc;
  }, {});
};
